\begin{document}

\maketitle

\begin{Grammar}

\begin{Rules}{File}
\RHS{\LISTGRP{\nt{Decl} \sym{;}}}
\end{Rules}

\begin{Rules}{Decl}
\RHS{\sym{%name} \kw{ID}}
\RHS{\sym{%start} \kw{ID}}
\RHS{\sym{%entry} \nt{IDList}}
\RHS{\sym{%keywords} \nt{SymList}}
\RHS{\sym{%defs} \nt{Code}}
\RHS{\sym{%tokens} \OPTGRP{\sym{:}} \nt{ConstrList}}
\RHS{\sym{%import} \kw{STRING} \OPTGRP{\sym{%dropping} \nt{SymList}}}
\RHS{\sym{%refcell} \kw{ID} \sym{:} \nt{Ty} \sym{=} \nt{Code}}
\RHS{\sym{%nonterms} \OPTGRP{\sym{:}} \nt{TyAnn} \LISTGRP{\sym{$|$} \nt{TyAnn}}}
\RHS{\kw{ID} \OPTGRP{\nt{Formals}} \sym{:} \nt{AltList}}
\end{Rules}

\begin{Rules}{IDList}
\RHS{\nt{ID'} \LISTGRP{\sym{,} \nt{ID'}}}
\end{Rules}

\begin{Rules}{ID'}
\RHS{\kw{ID}}
\end{Rules}

\begin{Rules}{ConstrList}
\RHS{\nt{Constr} \LISTGRP{\sym{$|$} \nt{Constr}}}
\end{Rules}

\begin{Rules}{Constr}
\RHS{\kw{ID} \OPTGRP{\sym{of} \nt{Ty}} \OPTGRP{\nt{Abbrev}}}
\end{Rules}

\begin{Rules}{Abbrev}
\RHS{\sym{(} \kw{STRING} \sym{)}}
\end{Rules}

\begin{Rules}{SymList}
\RHS{\nt{Symbol} \LISTGRP{\sym{,} \nt{Symbol}}}
\end{Rules}

\begin{Rules}{TyAnn}
\RHS{\kw{ID} \nt{\GRP{\sym{of} \nt{Ty} $\;|\;$ }}}
\end{Rules}

\begin{Rules}{Ty}
\RHS{\sym{\{} \nt{RowList} \sym{\}}}
\RHS{\sym{\{} \sym{\}}}
\RHS{\nt{TyApp}}
\end{Rules}

\begin{Rules}{TyApp}
\RHS{\sym{(} \nt{Ty} \LISTGRP{\sym{,} \nt{Ty}} \sym{)} \nt{QualID}}
\RHS{\nt{TyProd} \nt{\GRP{\nt{QualID} $\;|\;$ }}}
\end{Rules}

\begin{Rules}{TyProd}
\RHS{\nt{TyFun} \LISTGRP{\sym{*} \nt{TyFun}}}
\end{Rules}

\begin{Rules}{TyFun}
\RHS{\nt{TyAtom} \LISTGRP{\sym{$-$$>$} \nt{TyAtom}}}
\end{Rules}

\begin{Rules}{TyAtom}
\RHS{\kw{TYVAR}}
\RHS{\sym{(} \nt{Ty} \sym{)}}
\RHS{\nt{QualID}}
\end{Rules}

\begin{Rules}{RowList}
\RHS{\nt{Row} \LISTGRP{\sym{,} \nt{Row}}}
\end{Rules}

\begin{Rules}{Row}
\RHS{\nt{Label} \sym{:} \nt{Ty}}
\end{Rules}

\begin{Rules}{Label}
\RHS{\kw{ID}}
\RHS{\kw{INT}}
\end{Rules}

\begin{Rules}{QualID}
\RHS{\kw{ID}}
\RHS{\kw{IDDOT} \nt{QualID}}
\end{Rules}

\begin{Rules}{Formals}
\RHS{\sym{(} \kw{ID} \LISTGRP{\sym{,} \kw{ID}} \sym{)}}
\end{Rules}

\begin{Rules}{AltList}
\RHS{\nt{Alt} \LISTGRP{\sym{$|$} \nt{Alt}}}
\end{Rules}

\begin{Rules}{Alt}
\RHS{\OPTGRP{\sym{%try}} \LISTGRP{\nt{NamedItem}} \OPTGRP{\sym{%where} \nt{Code}} \OPTGRP{\sym{=$>$} \nt{Code}}}
\end{Rules}

\begin{Rules}{NamedItem}
\RHS{\OPTGRP{\kw{ID} \sym{=}} \nt{Item}}
\end{Rules}

\begin{Rules}{Item}
\RHS{\nt{PrimItem} \nt{\GRP{\sym{*} $\;|\;$ \sym{+} $\;|\;$ \sym{?} $\;|\;$ }}}
\end{Rules}

\begin{Rules}{PrimItem}
\RHS{\nt{Symbol} \OPTGRP{\sym{@} \nt{Code}}}
\RHS{\sym{(} \nt{AltList} \sym{)}}
\end{Rules}

\begin{Rules}{Symbol}
\RHS{\kw{ID}}
\RHS{\kw{STRING}}
\end{Rules}

\begin{Rules}{Code}
\RHS{\kw{CODE}}
\end{Rules}

\end{Grammar}

\end{document}
